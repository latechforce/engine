Automation with run TypeScript code action and externals

index.ts
import { App, type AppSchema, type CodeContext } from '@latechforce/engine';

const externals = {
  customFunction: () => {
    return 'Hello, world!';
  },
};

const schema: AppSchema = {
  automations: [
    {
      name: 'run-typescript',
      trigger: {
        service: 'http',
        event: 'post',
        path: 'run-typescript',
      },
      actions: [
        {
          service: 'code',
          action: 'run-typescript',
          name: 'runTypescriptCode',
          code: String(function (context: CodeContext<{}, typeof externals>) {
            const { customFunction } = context.externals;
            return { message: customFunction() };
          }),
        },
      ],
    },
  ],
};

await new App({ externals }).start(schema);
