Automation with HTTP GET trigger and respond immediately

index.ts
import { App, type AppSchema } from '@latechforce/engine';

const schema: AppSchema = {
  automations: [
    {
      id: 1,
      name: 'get',
      trigger: {
        service: 'http',
        event: 'get',
        getHttp: {
          path: 'get',
          respondImmediately: true,
        },
      },
      actions: [
        {
          name: 'response',
          service: 'http',
          action: 'response',
          responseHttp: {
            body: {
              message: 'Hello, world!',
            },
          },
        },
      ],
    },
  ],
};

await new App().start(schema);
