Automation with HTTP response action and static body

```typescript title="index.ts"
import { App, type AppSchema } from '@latechforce/engine'

const schema: AppSchema = {
  automations: [
    {
      name: 'response',
      trigger: {
        service: 'http',
        event: 'post',
        path: 'response',
      },
      actions: [
        {
          name: 'response',
          service: 'http',
          action: 'response',
          body: {
            message: 'Hello, world!',
          },
        },
      ],
    },
  ],
}

await new App().start(schema)
```
