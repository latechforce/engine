Automation with run TypeScript code action

```typescript title="index.ts"
import { App, type AppSchema } from '@latechforce/engine';

const schema: AppSchema = {
  automations: [
    {
      id: 1,
      name: 'run-typescript',
      editUrl:
        'https://github.com/latechforce/engine/blob/main/example/automation/admin.ts',
      trigger: {
        service: 'http',
        event: 'post',
        postHttp: {
          path: '/run-typescript',
        },
      },
      actions: [
        {
          service: 'code',
          action: 'run-typescript',
          name: 'runTypescriptCode',
          runTypescriptCode: {
            code: String(function () {
              return { message: 'Hello, world!' };
            }),
          },
        },
      ],
    },
  ],
};

await new App().start(schema);
```
